    <script>
        function showNotification(text) {
            const box = document.getElementById('notification');
            const boxText = document.getElementById('notif-text');
            boxText.innerText = text;
            box.style.display = 'block';
            setTimeout(() => { box.style.display = 'none'; }, 6000);
        }

        function scrollToDaftar() {
            document.getElementById('daftar').scrollIntoView({ behavior: 'smooth' });
        }

        // WhatsApp Registration
        document.getElementById('suayan-final-reg').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const adminNumber = "6289525140312";
            const team = document.getElementById('team_name').value;
            const cap = document.getElementById('captain_name').value;
            const wa = document.getElementById('whatsapp').value;
            const lineup = document.getElementById('members').value;

            const message = `*REGISTRASI SUAYAN CHAMPIONSHIP*%0A` +
                            `*EDISI 16 GRUP ELIT - 5 PEMAIN*%0A` +
                            `------------------------------------%0A` +
                            `⚔️ *Nama Grup:* ${team.toUpperCase()}%0A` +
                            `👑 *Kapten:* ${cap}%0A` +
                            `📱 *WhatsApp Kapten:* ${wa}%0A` +
                            `------------------------------------%0A` +
                            `👥 *Lineup Tim:*%0A${lineup.replace(/\n/g, '%0A')}%0A` +
                            `------------------------------------%0A` +
                            `_Masa Registrasi: 30 Mar - 3 Mei 2026_`;

            const btn = e.target.querySelector('button');
            const originalText = btn.innerText;
            btn.innerText = "OPENING PORTAL WHATSAPP...";
            btn.disabled = true;

            setTimeout(() => {
                showNotification("Mengalihkan ke WhatsApp Admin... Silakan kirim data Anda.");
                window.open(`https://wa.me/${adminNumber}?text=${message}`, '_blank');
                btn.innerText = originalText;
                btn.disabled = false;
                e.target.reset();
            }, 1200);
        });

        // Animation on scroll
        const observerOptions = { threshold: 0.1 };
        const observer = new IntersectionObserver((entries) => {
            entries.forEach(entry => {
                if (entry.isIntersecting) {
                    entry.target.style.opacity = "1";
                    entry.target.style.transform = "translateY(0)";
                }
            });
        }, observerOptions);

        document.querySelectorAll('.mlbb-card').forEach(el => {
            el.style.opacity = "0";
            el.style.transform = "translateY(50px)";
            el.style.transition = "all 0.8s cubic-bezier(0.165, 0.84, 0.44, 1)";
            observer.observe(el);
        });
    </script>
